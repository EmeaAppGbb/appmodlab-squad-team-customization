anonymization_utilities:

  fake_patient_generator:
    name:
      pattern:  "Test {FirstName} {LastName}"
      example:  "Test John Smith"        "Test Mary Johnson"

    mrn:
      pattern:  "MRN-T{7_digits}"
      example:  "MRN-T1234567"           # "T" marker = test data

    ssn:
      pattern:  "000-00-{4_digits}"
      example:  "000-00-1234"            # 000-00 = clearly fake

    phone:
      pattern:  "(555) 01{2_digits}"
      example:  "(555) 0145"             # 555-01XX = reserved range

    email:
      pattern:  "patient.test{N}@example.com"
      example:  "patient.test1234@example.com"

    address:
      example:  "123 Test Street, Anytown, NY 10001"

  validation_checks:
    - name:     "No real SSNs"
      pattern:  '^(?!000)(?!666)(?!9)\d{3}-(?!00)\d{2}-(?!0000)\d{4}$'
      severity: critical
      message:  "Test data contains a potentially real SSN. Use 000-00-XXXX."

    - name:     "No real names"
      check:    "name must start with 'Test '"
      severity: high

    - name:     "Test MRN prefix"
      pattern:  '^MRN-T\d{7}$'
      severity: high
