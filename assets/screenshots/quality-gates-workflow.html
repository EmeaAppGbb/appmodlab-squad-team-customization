.github/workflows/squad-quality-gates.yml ⚙️ CI/CD 
name: SQUAD Quality Gates

on:
  pull_request:
    branches: [ main, develop ]
    paths:    [ 'medbook/**' ]
  push:
    branches: [ main ]

jobs:

  # ━━━ Job 1: HIPAA Compliance ━━━
  hipaa-compliance:
    name: HIPAA Compliance Check
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: github/copilot-cli-action@v1
      - run:  copilot squad run --agent hipaa-compliance --check compliance_review
        continue-on-error: false
      - name: Check for Critical Violations
        run:  VIOLATIONS=$(jq '.violations | length' compliance-report.json)
              if [ "$VIOLATIONS" -gt 0 ]; then exit 1; fi

  # ━━━ Job 2: Terminology Validation ━━━
  terminology-validation:
    name: Healthcare Terminology Validation
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: github/copilot-cli-action@v1
      - run:  copilot squad run --agent terminology --check terminology_validation

  # ━━━ Job 3: Test Data Anonymization ━━━
  test-data-anonymization:
    name: Validate Test Data Anonymization
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: github/copilot-cli-action@v1
      - run:  copilot squad run --agent anonymizer --check test_data_validation
      - name: Check for Real PHI in Tests
        run:  PHI=$(jq '.phi_in_tests | length' anonymization-report.json)
              if [ "$PHI" -gt 0 ]; then exit 1; fi
