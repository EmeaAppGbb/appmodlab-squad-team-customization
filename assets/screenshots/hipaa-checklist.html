hipaa_compliance_checklist:

  phi_exposure:
    - rule:     "No PHI in log.Info, log.Debug, log.Error"
      severity: critical
      pattern:  'log\.(Info|Debug|Error|Printf).*patient\.(Name|FirstName|LastName|SSN|DOB|MRN)'
      suggestion: "Use patient ID instead: log.Info(\"Patient action\", \"patient_id\", patient.ID)"

    - rule:     "No PHI in HTTP error responses"
      severity: critical
      pattern:  'c\.JSON.*patient\.(Name|FirstName|LastName|SSN|DOB)'

    - rule:     "No PHI in metrics or telemetry"
      severity: critical
      reason:  "PHI must not be sent to observability platforms"

    - rule:     "No SSN in API responses"
      severity: critical

  encryption:
    - rule:     "Database connections must use SSL"
      severity: critical
      check:    "connection string contains sslmode=require"

    - rule:     "API must enforce TLS 1.2+"
      severity: critical

    - rule:     "PHI fields should be encrypted at rest"
      severity: high
      applies_to: [ssn, diagnosis_code, notes]

  audit_logging:
    - rule:     "PHI access must be logged"
      severity: high
      suggestion: "Add: audit.Log(userID, \"PHI_ACCESS\", patientID)"

  access_control:
    - rule:     "PHI endpoints require authentication"
      severity: critical
      check:    "middleware includes auth.RequireToken"

    - rule:     "Minimum necessary data returned"
      severity: medium
      suggestion: "Remove SSN, address from non-essential responses"

validation_levels:
  critical:  action: block_merge       notification: immediate
  high:      action: require_review   notification: within_24h
  medium:    action: warn             notification: next_review
